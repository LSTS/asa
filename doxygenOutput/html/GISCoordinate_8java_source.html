274  private void direct(double latRad, double lonRad, double dirRad,
275  double distNM) throws Exception {
276  double EPS = 0.00000000005;
277  double dlon, lat, lon;
278  if ((Math.abs(Math.cos(latRad)) < EPS)
279  && !(Math.abs(Math.sin(dirRad)) < EPS)) {
280  throw new Exception(
281  "Only N-S courses are meaningful, starting at a pole!");
282  }
283 
284  lat = Math.asin(Math.sin(latRad) * Math.cos(distNM) + Math.cos(latRad)
285  * Math.sin(distNM) * Math.cos(dirRad));
286  if (Math.abs(Math.cos(lat)) < EPS) {
287  lon = 0; // endpoint a pole
288  } else {
289  dlon = Math.atan2(
290  Math.sin(dirRad) * Math.sin(distNM) * Math.cos(latRad),
291  Math.cos(distNM) - Math.sin(latRad) * Math.sin(lat));
292  lon = mod(lonRad - dlon + Math.PI, 2 * Math.PI) - Math.PI;
293  }
294 
295  _lat = lat * (180 / Math.PI);
296  _lon = lon * (180 / Math.PI);
297  }

299  private void direct_ell(double glat1, double glon1, double faz, double s,
300  Ellipsoid ellipse) throws Exception {
301  // glat1 initial geodetic latitude in radians N positive
302  // glon1 initial geodetic longitude in radians E positive
303  // faz forward azimuth in radians
304  // s distance in units of a (=nm)
305 
306  double EPS = 0.00000000005;
307  double r = 0, tu = 0, sf = 0, cf = 0, b = 0, cu = 0, su = 0, sa = 0, c2a = 0, x = 0, c = 0, d = 0, y = 0, sy = 0, cy = 0, cz = 0, e = 0, a = 0;
308  double glat2, glon2, f;
309  @SuppressWarnings("unused")
310  double baz;
311 
312  if ((Math.abs(Math.cos(glat1)) < EPS)
313  && !(Math.abs(Math.sin(faz)) < EPS)) {
314  throw new Exception(
315  "Only N-S courses are meaningful, starting at a pole!");
316  }
317 
318  a = ellipse.a;
319  f = 1 / ellipse.invf;
320  r = 1 - f;
321  tu = r * Math.tan(glat1);
322  sf = Math.sin(faz);
323  cf = Math.cos(faz);
324  if (cf == 0) {
325  b = 0.;
326  } else {
327  b = 2. * Math.atan2(tu, cf);
328  }
329  cu = 1. / Math.sqrt(1 + tu * tu);
330  su = tu * cu;
331  sa = cu * sf;
332  c2a = 1 - sa * sa;
333  x = 1. + Math.sqrt(1. + c2a * (1. / (r * r) - 1.));
334  x = (x - 2.) / x;
335  c = 1. - x;
336  c = (x * x / 4. + 1.) / c;
337  d = (0.375 * x * x - 1.) * x;
338  tu = s / (r * a * c);
339  y = tu;
340  c = y + 1;
341  while (Math.abs(y - c) > EPS) {
342  sy = Math.sin(y);
343  cy = Math.cos(y);
344  cz = Math.cos(b + y);
345  e = 2. * cz * cz - 1.;
346  c = y;
347  x = e * cy;
348  y = e + e - 1.;
349  y = (((sy * sy * 4. - 3.) * y * cz * d / 6. + x) * d / 4. - cz)
350  * sy * d + tu;
351  }
352 
353  b = cu * cy * cf - su * sy;
354  c = r * Math.sqrt(sa * sa + b * b);
355  d = su * cy + cu * sy * cf;
356  glat2 = modlat(Math.atan2(d, c));
357  c = cu * cy - su * sy * cf;
358  x = Math.atan2(sy * sf, c);
359  c = ((-3. * c2a + 4.) * f + 4.) * c2a * f / 16.;
360  d = ((e * cy * c + cz) * sy * c + y) * sa;
361  glon2 = modlon(glon1 + x - (1. - c) * d * f); // fix date line problems
362  baz = modcrs(Math.atan2(sa, b) + Math.PI);
363  _lat = glat2 * (180 / Math.PI);
364  _lon = glon2 * (180 / Math.PI);
365  }

383  private double parselatlon(String str) throws Exception {
384  // Parse string in the format: dd:mm:ss.sssss
385  double ret = 0;
386  java.util.StringTokenizer tok = new java.util.StringTokenizer(str, ":");
387  if (tok.hasMoreTokens()) {
388  try {
389  ret += Integer.parseInt(tok.nextToken()); // degrees
390  } catch (NumberFormatException e) {
391  throw new Exception("Degrees must be an integer");
392  }
393  if (tok.hasMoreTokens()) {
394  int min = 0;
395  try {
396  min = Integer.parseInt(tok.nextToken());// minutes
397  } catch (NumberFormatException e) {
398  throw new Exception("minutes must be an integer");
399  }
400  if (min >= 60) {
401  throw new Exception("minutes must be less than 60");
402  }
403  ret += min / 60.0;
404  if (tok.hasMoreTokens()) {
405  double sec = 0;
406  try {
407  sec = Double.parseDouble(tok.nextToken());// seconds
408  } catch (NumberFormatException e) {
409  throw new Exception(
410  "minutes must be in decimal or integer format");
411  }
412  if (sec >= 60) {
413  throw new Exception("seconds must be less than 60");
414  }
415  ret += sec / 3600.0;
416  }
417  }
418  } else {
419  throw new Exception(str + " is not a valid coordinate format");
420  }
421  return ret;
422  }
